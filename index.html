 $('.space').mousemove(function (e) {
   var offset = $(this).offset();
   var relativeX = (e.pageX - offset.left);

   $('.space').hover(() => {
     $(this).css({
       'background-position': `${relativeX}px 350px, bottom left`
     });
   }, () => {
     $(this).css('background-position', 'center 150px, top left');
   });
   $('.space').trigger('mouseleave').trigger('mouseenter');
 });

 var fadeIn = (idx) => {
   var varters = shuffle($('[class*="char"]', $intro[idx]));
   $($intro[idx]).show();
   $.each(varters, (i, el) => {
     setTimeout(() => ($(el).animate({ opacity: 1 }, 100)),
       (500 / varters.length * i) + (500)
     );
   });
 };

 var fadeOut = (idx) => {
   var varters = shuffle($('[class*="char"]', $intro[idx]));
   $.each(varters, (i, el) => {
     setTimeout(() => (
       $(el).animate({ opacity: 0 }, 100)), (500 / varters.length * i)
     );
   });
 };
